# Create a new Ethereum keypair and get its balance.


npm exec -- ts-node scripts/create-private-key.ts > user-data/example-private-key.txt

cat user-data/example-private-key.txt


cat user-data/example-private-key.txt | npm exec -- ts-node scripts/derive-address.ts > user-data/example-address.txt

cat user-data/example-address.txt


npm exec -- ts-node scripts/get-balance.ts --address-file user-data/example-address.txt
